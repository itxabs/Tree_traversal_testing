#include <iostream>
using namespace std;
class Node
{
public:
    int key;
    Node *right;
    Node *left;
Node(int data)
{
  key = data;
  left = NULL;
  right = NULL;
}
};
class Queue
{
private:
  int front;
  int rear;
  int size=10;
  Node *Q[10];
public:
  Queue()
{
  front = -1;
  rear = -1;
}
 void EnQueue(Node * x)
  {
    if (Isfull())
    {
      cout << "Queue is full:";
    }
    if (front == -1) 
    {
          front = 0;
          Q[front]=x;
    }
    else  
    Q[++rear] = x;
  }
  void deQueue()
  {
    if (IsEmtpy()) 
    {
      cout << "Queue is empty." << endl;
      return ;
    }
    else 
    {
      
        for (int i = front; i < rear; ++i) {
            Q[i] = Q[i + 1];
        }
        rear--;
        if (rear < front) {
            front = rear = -1; // Reset the queue when it becomes empty
        }
        
    }
  }
  bool Isfull()
  {
  return rear == size - 1;
  }
  bool IsEmtpy()
  {
    if(rear == -1 && front == -1)

  {
  return true;}
  else {
    return false;
  }
  }
  Node* getFront() {
    if (IsEmtpy()) {
      cout << "Queue is empty." << endl;
      return NULL;
    }
    else {
      return Q[front];

    }
  }

};
class Bt
{
public:
Node* root;
Bt()
{  root=NULL; }

void Inorder(Node* root)
{
  if(root!=NULL)
  {
    Inorder(root->left);
    cout<<root->key<<" ";
    Inorder(root->right);

  }
}
void Preorder(Node* root)
{
  if(root!=NULL)
  {
    cout<<root->key<<" ";
    Preorder(root->left);
    Preorder(root->right);

  }
}
void Postorder(Node* root)
{
  if(root!=NULL)
  {
    Postorder(root->left);
    Postorder(root->right);
    cout<<root->key<<" ";
  }
}

void Levelwise(Node* root)
{
  if(root==NULL)
  {
    return ;

  }
  Queue q;
    q.EnQueue(root);

    while(q.IsEmtpy()==false)
      { 
         Node* curr = q.getFront();
        //cout<<"value "<<curr->key;
        q.deQueue();
        cout<<curr->key<<" ";

        if(curr->left!=NULL)
        {
          q.EnQueue(curr->left);
        }
        if(curr->right!=NULL)
        {
          q.EnQueue(curr->right);
        }


      }
}

};
